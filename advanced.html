function bubbleSort(array) {
    const arr = [...array];           // Clone array to preserve original
    const n = arr.length;
    let comparisons = 0;
    let swaps = 0;
    
    for (let i = 0; i < n; i++) {
        let swapped = false;
        
        for (let j = 0; j < n - 1 - i; j++) {
            comparisons++;
            
            if (arr[j] > arr[j + 1]) {
                // Visual swap trigger
                [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
                swaps++;
                swapped = true;
            }
        }
        
        if (!swapped) break;          // Optimization: exit if sorted
    }
    
    return { sorted: arr, comparisons, swaps };
}

Time Complexity

O(n²)

Quadratic time complexity in average and worst cases. Each element must be compared with every other element, creating nested loops that scale exponentially with input size.

Space Complexity

O(n)

Linear space for the copied array. The algorithm operates with O(1) auxiliary space, but we maintain immutability by returning a new sorted array rather than mutating the input.

Best Case

O(n)

When the input array is already sorted, the optimized implementation detects this in a single pass and exits early, achieving linear time complexity.
